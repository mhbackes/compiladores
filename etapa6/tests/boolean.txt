bool bf:0;
bool bt:1;
bool br:0;

int main() {
    
    // AND
    output "AND TESTS \n\n"

    br = bf && bf
    if(br) 
        output "FALSE AND FALSE\n"

    br = bf && bt
    if(br)
        output "FALSE AND TRUE\n"

    br = bt && bt
    if(br)
        output "TRUE AND TRUE\n\n"

    // OR 
    output "OR TESTS \n\n"

    br = bf || bf
    if(br)
        output "FALSE OR FALSE\n"

    br = bf || bt
    if(br)
        output "FALSE OR TRUE\n"

    br = bt || bt
    if(br)
        output "TRUE OR TRUE\n\n"


    // BINARY OP TESTS
    output "LESS    \n\n"
    if(2 < 3)
        output "um pais da europa acertou\n"

    if(2 > 3)
        output "um pais da europa errou\n"

    if(2 > 2)
        output "um pais da europa errou\n"

    output "LE      \n\n"
    if(2 <= 3)
        output "um pais da europa acertou\n"

    if(2 <= 2)
        output "um pais da europa acertou\n"

    if(2 <= 1)
        output "um pais da europa errou\n"

    output "GREATER \n\n"
    
    output "GE      \n\n"

    output "EQ      \n\n"

    output "NE      \n\n" 

    // NOT/UNARY TESTS
    output "NOT     \n\n"

    return 0
};
